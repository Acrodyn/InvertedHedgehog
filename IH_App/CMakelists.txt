project(IH_App)

file(GLOB_RECURSE APP_SOURCES "src/*.cpp" "src/*.h")

add_executable(IH_App ${APP_SOURCES})

target_link_libraries(IH_App PRIVATE IH_Engine)

target_include_directories(IH_App PRIVATE
    $<TARGET_PROPERTY:IH_Engine,INTERFACE_INCLUDE_DIRECTORIES>
)

add_custom_command(TARGET IH_App POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    $<TARGET_FILE:IH_Engine> $<TARGET_FILE_DIR:IH_App>)
